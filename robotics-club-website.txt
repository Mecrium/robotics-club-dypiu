/**
 * Scroll Animation Handler
 * Detects when elements come into view and applies animations
 */
class ScrollAnimationHandler {
  constructor() {
    this.observerOptions = {
      threshold: 0.1,
      rootMargin: '0px 0px -100px 0px'
    };
    this.init();
  }

  init() {
    const observer = new IntersectionObserver((entries) => {
      entries.forEach(entry => {
        if (entry.isIntersecting) {
          this.animateElement(entry.target);
          observer.unobserve(entry.target);
        }
      });
    }, this.observerOptions);

    // Observe all elements with animation classes
    document.querySelectorAll(
      '.animate-on-scroll, ' +
      '.animate-slide-left, ' +
      '.animate-slide-right, ' +
      '.animate-zoom'
    ).forEach(el => observer.observe(el));
  }

  animateElement(element) {
    element.classList.add('visible');
  }
}

/**
 * Parallax Scrolling Effect
 * Creates depth effect by moving elements at different speeds
 */
class ParallaxEffect {
  constructor() {
    this.elements = document.querySelectorAll('[data-parallax]');
    this.init();
  }

  init() {
    if (this.elements.length === 0) return;

    window.addEventListener('scroll', () => this.update());
  }

  update() {
    const scrollY = window.pageYOffset;

    this.elements.forEach(element => {
      const speed = parseFloat(element.getAttribute('data-parallax')) || 0.5;
      const yOffset = scrollY * speed;
      element.style.transform = `translateY(${yOffset}px)`;
    });
  }
}

/**
 * Smooth Counter Animation
 * Animates numbers from 0 to target value
 */
class CounterAnimation {
  static animate(element, targetValue, duration = 2000) {
    const start = 0;
    const increment = targetValue / (duration / 16);
    let current = start;

    const timer = setInterval(() => {
      current += increment;
      if (current >= targetValue) {
        element.textContent = targetValue.toLocaleString();
        clearInterval(timer);
        return;
      }
      element.textContent = Math.floor(current).toLocaleString();
    }, 16);
  }

  static initializeAll() {
    const counters = document.querySelectorAll('[data-counter]');
    
    const observer = new IntersectionObserver((entries) => {
      entries.forEach(entry => {
        if (entry.isIntersecting) {
          const target = parseInt(entry.target.getAttribute('data-counter'));
          const duration = parseInt(entry.target.getAttribute('data-duration')) || 2000;
          
          this.animate(entry.target, target, duration);
          observer.unobserve(entry.target);
        }
      });
    }, { threshold: 0.5 });

    counters.forEach(counter => observer.observe(counter));
  }
}

/**
 * Typewriter Text Effect
 * Creates typing animation for text content
 */
class TypewriterEffect {
  constructor(element, options = {}) {
    this.element = element;
    this.text = element.textContent;
    this.speed = options.speed || 50;
    this.delay = options.delay || 0;
    this.loop = options.loop || false;
    this.onComplete = options.onComplete || null;

    this.init();
  }

  init() {
    setTimeout(() => this.start(), this.delay);
  }

  start() {
    this.element.textContent = '';
    let index = 0;

    const type = () => {
      if (index < this.text.length) {
        this.element.textContent += this.text.charAt(index);
        index++;
        setTimeout(type, this.speed);
      } else {
        if (this.onComplete) this.onComplete();
        if (this.loop) setTimeout(() => this.start(), 2000);
      }
    };

    type();
  }

  static initializeAll() {
    document.querySelectorAll('[data-typewriter]').forEach(element => {
      const speed = parseInt(element.getAttribute('data-typewriter')) || 50;
      const delay = parseInt(element.getAttribute('data-delay')) || 0;
      const loop = element.getAttribute('data-loop') === 'true';

      new TypewriterEffect(element, { speed, delay, loop });
    });
  }
}

/**
 * Glow Effect Handler
 * Creates neon glow effects on hover
 */
class GlowEffect {
  constructor() {
    this.elements = document.querySelectorAll('[data-glow]');
    this.init();
  }

  init() {
    this.elements.forEach(element => {
      element.addEventListener('mousemove', (e) => this.handleMouseMove(e));
      element.addEventListener('mouseleave', (e) => this.handleMouseLeave(e));
    });
  }

  handleMouseMove(event) {
    const element = event.target.closest('[data-glow]');
    const glowColor = element.getAttribute('data-glow') || '#00d4ff';
    const glowSize = parseInt(element.getAttribute('data-glow-size')) || 20;

    element.style.boxShadow = `0 0 ${glowSize}px ${glowColor}`;
  }

  handleMouseLeave(event) {
    event.target.closest('[data-glow]').style.boxShadow = '';
  }
}

/**
 * Staggered List Animation
 * Animates list items with delays
 */
class StaggeredAnimation {
  static animate(container, itemSelector = '.stagger-item', delay = 0.1) {
    const items = container.querySelectorAll(itemSelector);
    
    items.forEach((item, index) => {
      item.style.animation = `slideInUp 0.6s ease-out forwards`;
      item.style.animationDelay = `${index * delay}s`;
    });
  }

  static initializeAll() {
    document.querySelectorAll('[data-stagger]').forEach(container => {
      const itemSelector = container.getAttribute('data-stagger') || '.stagger-item';
      const delay = parseFloat(container.getAttribute('data-stagger-delay')) || 0.1;
      
      this.animate(container, itemSelector, delay);
    });
  }
}

/**
 * Fade In On Scroll
 * Fades in elements as they enter viewport
 */
class FadeInOnScroll {
  constructor() {
    this.elements = document.querySelectorAll('[data-fade-in]');
    this.init();
  }

  init() {
    const observer = new IntersectionObserver((entries) => {
      entries.forEach(entry => {
        if (entry.isIntersecting) {
          entry.target.style.opacity = '1';
          entry.target.style.transform = 'translateY(0)';
          observer.unobserve(entry.target);
        }
      });
    }, { threshold: 0.1 });

    this.elements.forEach(element => {
      element.style.opacity = '0';
      element.style.transform = 'translateY(20px)';
      element.style.transition = 'opacity 0.6s ease-out, transform 0.6s ease-out';
      observer.observe(element);
    });
  }
}

/**
 * Bounce Animation Handler
 */
class BounceAnimation {
  static init() {
    const bounceElements = document.querySelectorAll('[data-bounce]');
    
    bounceElements.forEach(element => {
      element.addEventListener('mouseenter', function() {
        this.style.animation = 'none';
        // Trigger reflow to restart animation
        void this.offsetWidth;
        this.style.animation = 'hoverLift 0.6s ease-out forwards';
      });

      element.addEventListener('mouseleave', function() {
        this.style.animation = 'none';
      });
    });
  }
}

/**
 * Scroll Progress Indicator
 * Shows reading progress at top of page
 */
class ScrollProgress {
  constructor() {
    this.progressBar = document.getElementById('scrollProgress');
    if (!this.progressBar) {
      this.createProgressBar();
    }
    this.init();
  }

  createProgressBar() {
    const bar = document.createElement('div');
    bar.id = 'scrollProgress';
    bar.style.cssText = `
      position: fixed;
      top: 0;
      left: 0;
      height: 3px;
      background: linear-gradient(90deg, #00d4ff, #b700ff);
      z-index: 9999;
      box-shadow: 0 0 10px rgba(0, 212, 255, 0.5);
    `;
    document.body.appendChild(bar);
    this.progressBar = bar;
  }

  init() {
    window.addEventListener('scroll', () => this.update());
  }

  update() {
    const windowHeight = document.documentElement.scrollHeight - document.documentElement.clientHeight;
    const scrolled = (window.pageYOffset / windowHeight) * 100;
    this.progressBar.style.width = scrolled + '%';
  }
}

/**

class TextSplitAnimation {
  static splitText(element) {
    const text = element.textContent;
    element.textContent = '';

    text.split('').forEach((char, index) => {

      span.textContent = char;
      span.style.display = 'inline-block';
      span.style.animation = `slideInUp 0.5s ease-out ${index * 0.05}s both`;
      element.appendChild(span);
    });
  }

  static initializeAll() {
    document.querySelectorAll('[data-split-text]').forEach(element => {
      this.splitText(element);
    });
  }
}
